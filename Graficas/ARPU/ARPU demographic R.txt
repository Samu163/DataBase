# --- INICIO DEL CÓDIGO ---

# 1. INSTALAR Y CARGAR PAQUETES
# install.packages("tidyverse") # Descomenta esta línea si no tienes el paquete 'tidyverse'
library(tidyverse)

# 2. CARGAR LOS DATOS
# ¡¡CAMBIA "tu_archivo_arpu_demo.csv" POR EL NOMBRE DE TU ARCHIVO!!
df_arpu_demo <- read.csv("tu_archivo_arpu_demo.csv")

# 3. VERIFICAR COLUMNAS (DEBUG)
# Confirma que ves "country" y "arpu"
print("Columnas cargadas en el archivo:")
print(names(df_arpu_demo))

# 4. PREPARAR LOS DATOS
# Filtramos filas con NA en 'arpu'
df_arpu_demo <- df_arpu_demo %>%
  filter(!is.na(arpu))


# 5. CREAR LA GRÁFICA DE BARRAS
if (nrow(df_arpu_demo) > 0) {
    
    # 1. Añadimos 'fill = country' dentro de aes()
    ggplot(data = df_arpu_demo, aes(x = reorder(country, -arpu), 
                                   y = arpu, 
                                   fill = country)) +
        
        # 2. Capa de Barras
        geom_bar(stat = "identity") +
        
        # 3. Capa de Texto (con 2 decimales para ARPU)
        geom_text(
          # Usamos round(..., 2) para mostrar valores como 1.50
          aes(label = round(arpu, 2)), 
          vjust = -0.5,
          size = 3.5
        ) +
        
        # 4. Títulos y etiquetas
        labs(
            title = "ARPU by Country",
            x = "Country",
            y = "ARPU"
        ) +
        
        # 5. Tema limpio
        theme_minimal() +
        
        # 6. Rotar etiquetas Y QUITAR LA LEYENDA
        theme(
            axis.text.x = element_text(
                angle = 45,
                hjust = 1,
                vjust = 1
            ),
            legend.position = "none" # Quitamos la leyenda de colores
        )
    
} else {
    print("Data frame is empty, has NAs, or was not loaded correctly.")
}
